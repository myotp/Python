html = urllib2.urlopen(URL_INDEX_PAGE).read()
print 'HTML:', html
soup = BeautifulSoup(html)
result_table = soup.find('table')
rows = result_table.tbody.findAll('tr')
row0 = rows[0]
tds = row0.findAll('td')
td0 = tds[0]
type (td0)
dir (td0)
str(td0.getText())
str(td0.text)
link = td0.find('a')
type(link)
str(link.get('href'))
